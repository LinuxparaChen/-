# 系统启动流程

* **init进程**

    ①init.rc
    ```
    import /init.${ro.zygote}.rc //导入init.zygote64.rc

    on early-init //初始化早期执行
        start ueventd //启动ueventd服务，已启动直接跳过。ueventd：驱动服务。

    on init //初始化阶段执行
        //创建系统目录，控制权限

    on late-init //初始化晚期执行
        //创建系统目录，控制权限。开启 core class。//ueventd、adbd、servicemanager、surfaceflinger、bootanim等服务
    ```
    init.zygote64.rc
    ```
    service zygote /system/bin/app_process64 -Xzygote /system/bin --zygote --start-system-server
    class main
    socket zygote stream 660 root system
    onrestart write /sys/android_power/request_state wake
    onrestart write /sys/power/state on
    onrestart restart media
    onrestart restart netd
    ```
    ②init.cpp
    ```c
    int main(int argc, char** argv) {
        
        ......
        //初始化信号处理。
        signal_handler_init();

        //加载默认启动属性
        //#define PROP_PATH_RAMDISK_DEFAULT  "/default.prop"
        property_load_boot_defaults();
        //开启属性服务
        start_property_service();

        //解析init.rc配置文件。
        init_parse_config_file("/init.rc");
        ......

        //waiting_for_exec 默认= false
        while (true) {
            if (!waiting_for_exec) {
                execute_one_command();//获取第一条服务
                restart_processes();//开启所有服务。
            }

            int timeout = -1;
            if (process_needs_restart) {
                timeout = (process_needs_restart - gettime()) * 1000;
                if (timeout < 0)
                    timeout = 0;
            }

            if (!action_queue_empty() || cur_action) {
                timeout = 0;
            }

            bootchart_sample(&timeout);

            epoll_event ev;
            int nr = TEMP_FAILURE_RETRY(epoll_wait(epoll_fd, &ev, 1, timeout));
            if (nr == -1) {
                ERROR("epoll_wait failed: %s\n", strerror(errno));
            } else if (nr == 1) {
                ((void (*)()) ev.data.ptr)();
            }
        }

        return 0;
    }
    ```

    ②AndroidRuntime::start函数
    ```
    void AndroidRuntime::start(const char* className, const Vector<String8>& options, bool zygote)
    {
        ......
        /* start the virtual machine */
        JniInvocation jni_invocation;
        jni_invocation.Init(NULL);
        JNIEnv* env;
        if (startVm(&mJavaVM, &env, zygote) != 0) {
            return;
        }
        onVmCreated(env);

        /*
        * Register android functions.
        */
        if (startReg(env) < 0) {
            ALOGE("Unable to register all android natives\n");
            return;
        }

        ......//组织main函数的参数。

        //jni 反射调用main函数。
        char* slashClassName = toSlashClassName(className);
        jclass startClass = env->FindClass(slashClassName);
        if (startClass == NULL) {
            ALOGE("JavaVM unable to locate class '%s'\n", slashClassName);
            /* keep going */
        } else {
            jmethodID startMeth = env->GetStaticMethodID(startClass, "main",
                                                        1090            "([Ljava/lang/String;)V");
            if (startMeth == NULL) {
                ALOGE("JavaVM unable to find main() in '%s'\n", className);
                /* keep going */
            } else {
                env->CallStaticVoidMethod(startClass, startMeth, strArray);
            }
        }
        free(slashClassName);

        ALOGD("Shutting down VM\n");
        if (mJavaVM->DetachCurrentThread() != JNI_OK)
            ALOGW("Warning: unable to detach main thread\n");
        if (mJavaVM->DestroyJavaVM() != 0)
            ALOGW("Warning: VM did not shut down cleanly\n");
    }
    ```
